%modificadorTipo ("daniel"|"dalila"|"paulo"|"joaoarthur"|"danielMendes")

break
  : break ';'

case 
  : case expr ':' stmt ';'
  | default ':' stmt ';'

continue
  : continue ';'

conditional
  : if 'baixo' expr 'alto' elseif
  | if 'baixo' expr 'alto' elseif else 'baixo' expr 'alto'

definicaoVariavel 
  : modificadoresVariaveis id '=' expr ';'

definicaoFuncao 
  : modificadoresFuncao id 'baixo'

do 
  : do 'bonito' stmt 'feio' while 'baixo' expr 'alto' ';'

elseif
  : else condicional
  | <nulo>

expr
  : expr '*' expr
  | expr '/' expr
  | expr '^' expr
  | expr '!=' expr
  | expr '&&' expr
  | expr '&' expr
  | expr '==' expr
  | expr '>=' expr
  | expr '>' expr
  | expr '+' expr
  | expr '<=' expr
  | expr '<' expr
  | expr '-' expr
  | expr '%' expr
  | expr '|' expr
  | expr '||' expr
  | expr '<<' expr
  | expr '>>' expr
  | expr 'endoidou' expr
  | '(' expr ')'
  | term

term : letter | id | positive | negative | decimal

for 
  : for 'baixo' stmt ';' expr ';' stmt 'alto' 'bonito' stmt 'feio' ';'

goto
  : goto label ';'

label
  : id ':'

modificadoresFuncao
  : modificadoresVariaveis modificadoresVariaveis
  | modificadorSinal
  | modificadorTamanho
  | modificadorTipo
  | <vazio>

modificadoresVariaveis
  : modificadoresVariaveis modificadoresVariaveis
  | modificadorSinal
  | modificadorEspecial
  | modificadorSomenteLeitura
  | modificadorTamanho
  | modificadorTipo
  | <vazio>

parametros
  :  parametros ',' parametros
  | modificadoresVariaveis id
  | <vazio>

return
  : return expr ';'
  | return ';'

sizeof
  : sizeof 'baixo' id 'alto' ';'

stmt 
  : id '=' expr ';'
  | id '=' id ';'
  | stmt ';' stmt
  | <nulo>

switch 
  : switch '(' id ')' '{' variosCase '}'

variosCase
  : variosCase variosCase ';'
  | case ';'
  | <nulo>

while
  : while '(' expr ')' '{' stmt '}'
