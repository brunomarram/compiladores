%modificadorTipo ("daniel"|"dalila"|"paulo"|"joaoarthur"|"danielMendes")

break
  : break ';'

case 
  : case expr ':' stmt ';'
  | default ':' stmt ';'

continue
  : continue ';'

conditional
  : if 'baixo' expr 'alto' elseif
  | if 'baixo' expr 'alto' elseif else 'baixo' expr 'alto'

definicaoVariavel 
  : modificadoresVariaveis id '=' expr ';'

definicaoFuncao 
  : modificadoresFuncao id 'baixo'

do 
  : do 'bonito' stmt 'feio' while 'baixo' expr 'alto' ';'

elseif
  : else condicional
  | <nulo>

expr
  : expr '*' expr
  | expr '/' expr
  | expr '^' expr
  | expr '!=' expr
  | expr '&&' expr
  | expr '&' expr
  | expr '==' expr
  | expr '>=' expr
  | expr '>' expr
  | expr '+' expr
  | expr '<=' expr
  | expr '<' expr
  | expr '-' expr
  | expr '%' expr
  | expr '|' expr
  | expr '||' expr
  | expr '<<' expr
  | expr '>>' expr
  | expr 'endoidou' expr
  | term

term : letter | id | positive | negative | decimal

for 
  : for 'baixo' stmt ';' expr ';' stmt 'alto' 'bonito' stmt 'feio' ';'

goto
  : goto label ';'

label
  : id ':'

modificadoresFuncao
  : modificadoresVariaveis modificadoresVariaveis
  | modificadorSinal
  | modificadorTamanho
  | modificadorTipo
  | <vazio>

modificadoresVariaveis
  : modificadoresVariaveis modificadoresVariaveis
  | modificadorSinal
  | modificadorEspecial
  | modificadorSomenteLeitura
  | modificadorTamanho
  | modificadorTipo
  | <vazio>

parametros
  :  parametros ',' parametros
  | modificadoresVariaveis id
  | <vazio>

return
  : return stmt ';'
  | return ';'

sizeof
  : sizeof 'baixo' id 'alto' ';'

stmt 
  : id '=' id
  | id '+=' id
  | id '-=' id
  | stmt ';' stmt

switch 
  : switch 'baixo' id 'alto' 'bonito' variosCase 'feio'

variosCase
  : variosCase variosCase ';'
  | case ';'
  | <nulo>

while
  : while 'baixo' expr 'alto' 'bonito' stmt 'feio'
