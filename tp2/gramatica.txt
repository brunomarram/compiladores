%modificadorTipo ("daniel"|"dalila"|"paulo"|"joaoarthur"|"danielMendes")

break
  : break ';'

case 
  : case expr ':' stmt ';'
  | default ':' stmt ';'

continue
  : continue ';'

conditional
  : if abreExpressao expr fechaExpressao elseif
  | if abreExpressao expr fechaExpressao elseif else abreExpressao expr fechaExpressao

definicaoVariavel 
  : modificadoresVariaveis id igual expr ';'

definicaoFuncao 
  : modificadoresFuncao id abreExpressao

do 
  : do abreEscopo stmt fechaEscopo while abreExpressao expr fechaExpressao ';'

elseif
  : else condicional
  | <nulo>

expr
  : expr asterisco expr
  | expr barra expr
  | expr chapeu expr
  | expr diferente expr
  | expr eDouble expr
  | expr eLogico expr
  | expr igual expr
  | expr igualIgual expr
  | expr maiorOuIgual expr
  | expr maiorQue expr
  | expr mais expr
  | expr menorOuIgual expr
  | expr menorQue expr
  | expr menos expr
  | expr percentual expr
  | expr pipe expr
  | expr pipeDouble expr
  | expr shiftLeft expr
  | expr shiftRight expr
  | expr operadorDoido expr
  | '(' expr ')'
  | term

term : letter | id | positive | negative | decimal

for 
  : for abreExpressao stmt ';' expr ';' stmt fechaExpressao abreEscopo stmt fechaEscopo ';'

goto
  : goto label ';'

label
  : id ':'

modificadoresFuncao
  : modificadoresVariaveis modificadoresVariaveis
  | modificadorSinal
  | modificadorTamanho
  | modificadorTipo
  | <vazio>

modificadoresVariaveis
  : modificadoresVariaveis modificadoresVariaveis
  | modificadorSinal
  | modificadorEspecial
  | modificadorSomenteLeitura
  | modificadorTamanho
  | modificadorTipo
  | <vazio>

parametros
  :  parametros ',' parametros
  | modificadoresVariaveis id
  | <vazio>

return
  : return expr ';'
  | return ';'

sizeof
  : sizeof abreExpressao id fechaExpressao ';'

stmt 
  : id igual expr ';'
  | id igual id ';'
  | stmt ';' stmt
  | while
  | for
  | switch
  | goto
  | break
  | continue conditional,definicaoVariavel,definicaoFuncao

switch 
  : switch abreExpressao id fechaExpressao abreEscopo variosCase fechaEscopo

variosCase
  : variosCase variosCase ';'
  | case ';'
  | <nulo>

while
  : while abreExpressao expr fechaExpressao abreEscopo stmt fechaEscopo
